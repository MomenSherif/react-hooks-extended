useMapState

Aboutâ€‹

Manage state in a key value pair map easier with useMapState.

Importâ€‹

import { useMapState } from 'react-hooks-extended';

Baisc exampleâ€‹

  • JavaScript
  • TypeScript
function Example() {
  const [state, { set, setMultiple, remove }] = useMapState({
    loading: true,
    data: null,
    error: null,
  });

  return (
    <div>
      <pre>{JSON.stringify(state, null, 2)}</pre>
      <button onClick={() => set('loading', true)}>Loading | true</button>
      <button onClick={() => set('loading', false)}>Loading | false</button>
      <button onClick={() => set('loading', s => !s)}>Loading | toggle</button>
      <button
        onClick={() =>
          setMultiple({
            loading: false,
            data: [1, 2, 3],
          })
        }
      >
        Set data & loading
      </button>
      <button onClick={() => remove('data')}>Remove data</button>
    </div>
  );
}
function Example() {
  const [state, { set, setMultiple, remove }] = useMapState<{
    loading: boolean;
    data: number[] | null;
    error: string | null;
  }>({
    loading: true,
    data: null,
    error: null,
  });

  return (
    <div>
      <pre>{JSON.stringify(state, null, 2)}</pre>
      <button onClick={() => set('loading', true)}>Loading | true</button>
      <button onClick={() => set('loading', false)}>Loading | false</button>
      <button onClick={() => set('loading', s => !s)}>Loading | toggle</button>
      <button
        onClick={() =>
          setMultiple({
            loading: false,
            data: [1, 2, 3],
          })
        }
      >
        Set data & loading
      </button>
      <button onClick={() => remove('data')}>Remove data</button>
    </div>
  );
}

ArgumentTypeDescriptionRequiredDefault value

Returnâ€‹

Array with a stateful value and methods to update the map.

Return valueTypeDescription
stateS extends objectCurrent state
methodsobjectHelper methods to update map state
methods.set (key: keyof S, value: S|(prevState: S) => S) => void Set a key value pair in map
methods.remove(key: keyof S) => voidRemove a key value pair in map
methods.has(key: keyof S) => booleanIf key exists in map
methods.setMultiple(value: Partial<S>|(prevState: S) => Partial<S>) => voidset multiple key value pair in map
methods.removeMultiple(...keys: (keyof S)[]) => voidRemove multiple key value pair in map
methods.removeAllFunctionRemove all key value pair in map

function Playground() {
  const [state, { set, setMultiple, remove, removeMultiple, removeAll, has }] =
    useMapState({
      loading: true,
      data: null,
      error: null,
    });


  return (
    <div>
      <pre>{JSON.stringify(state, null, 2)}</pre>
      <p>has loading: {String(has('loading'))}</p>
      <button onClick={() => set('loading', true)}>Loading | true</button>
      <button onClick={() => set('loading', false)}>Loading | false</button>
      <button onClick={() => set('loading', s => !s)}>Loading | toggle</button>
      <button
        onClick={() =>
          setMultiple({
            loading: false,
            data: [1, 2, 3],
          })
        }
      >
        Set data & loading
      </button>
      <button onClick={() => remove('data')}>Remove data</button>
      <button onClick={removeAll}>Remove all</button>
      <button onClick={() => removeMultiple('loading', 'error')}>
        Remove loading & error
      </button>
    </div>
  );
}
